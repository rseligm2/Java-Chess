696  virtual QCString trWriteList(int numEntries)
697  {
698  QCString result;
699  int i;
700  // the inherits list contain `numEntries' classes
701  for (i=0;i<numEntries;i++)
702  {
703  // use generateMarker to generate placeholders for the class links!
704  result+=generateMarker(i); // generate marker for entry i in the list
705  // (order is left to right)
706 
707  if (i!=numEntries-1) // not the last entry, so we need a separator
708  {
709  if (i<numEntries-2) // not the fore last entry
710  result+=", ";
711  else // the fore last entry
712  result+=" e ";
713  }
714  }
715  return result;
716  }
