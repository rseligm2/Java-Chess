184 template<class T> class CompAccept : public DocNode
185 {
186  public:
187  CompAccept() { m_children.setAutoDelete(TRUE); }
188  void accept(DocVisitor *v)
189  {
190  T *obj = dynamic_cast<T *>(this);
191  v->visitPre(obj);
192  QListIterator<DocNode> cli(m_children);
193  DocNode *n;
194  for (cli.toFirst();(n=cli.current());++cli) n->accept(v);
195  v->visitPost(obj);
196  }
197  const QList<DocNode> &children() const { return m_children; }
198  QList<DocNode> &children() { return m_children; }
199  QString::Direction getTextDir(int nodeIndex) const
200  {
201  unsigned char resultDir = QString::DirNeutral;
202  for (uint i = nodeIndex; i < m_children.count(); i++)
203  {
204  DocNode* node = m_children.at(i);
205  QString::Direction nodeDir = node->getTextDir();
206  resultDir |= nodeDir;
207  if (resultDir == QString::DirMixed)
208  return QString::DirMixed;
209  }
210  return static_cast<QString::Direction>(resultDir);
211  }
212  QString::Direction getTextBasicDir(int nodeIndex) const
213  {
214  for (uint i = nodeIndex; i < m_children.count(); i++)
215  {
216  DocNode* node = m_children.at(i);
217  QString::Direction nodeDir = node->getTextBasicDir();
218  if (nodeDir != QString::DirNeutral)
219  return nodeDir;
220  }
221  return QString::DirNeutral;
222  }
223  virtual QString::Direction getTextDir() const
224  {
225  return getTextDir(0);
226  }
227  virtual QString::Direction getTextBasicDir() const
228  {
229  return getTextBasicDir(0);
230  }
231 
232  protected:
233  QList<DocNode> m_children;
234 };
